SELECT o.orderNumber, o.orderDate, o.customerNumber, SUM(CASE WHEN (od.quantityOrdered * od.priceEach) > 7000 THEN 1 ELSE 0 END) righeconValoreMag7000 FROM orders o, orderdetails od WHERE od.orderNumber=o.orderNumber GROUP BY o.orderNumber, o.orderDate, o.customerNumber HAVING SUM(CASE WHEN (od.quantityOrdered * od.priceEach) > 7000 THEN 1 ELSE 0 END);
